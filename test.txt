import ssl

from langchain_core.prompts import PromptTemplate

from config import settings

# ── SSL Fix ────────────────────────────────────────────────────────────────

if not settings.ssl_verify:
    ssl._create_default_https_context = ssl._create_unverified_context

    _orig_client_init = httpx.Client.__init__

    def _no_verify_client_init(self, *args, **kwargs):
        kwargs["verify"] = False
        _orig_client_init(self, *args, **kwargs)

    httpx.Client.__init__ = _no_verify_client_init

    _orig_async_client_init = httpx.AsyncClient.__init__

    def _no_verify_async_client_init(self, *args, **kwargs):
        kwargs["verify"] = False
        _orig_async_client_init(self, *args, **kwargs)

    httpx.AsyncClient.__init__ = _no_verify_async_client_init

def get_llm() -> ChatMistralAI:
    return ChatMistralAI(

# ── 1. QnA ─────────────────────────────────────────────────────────────────

QNA_TEMPLATE = """
Answer the question in 3 sentences.

Question: {text}
"""

def run_qna(question: str) -> str:
    chain = PromptTemplate.from_template(QNA_TEMPLATE) | get_llm() | StrOutputParser()
    return chain.invoke({"text": question})

PRODUCT_QNA_TEMPLATE = """

def run_product_qna(product_title, bullet_points, description, question):
    chain = PromptTemplate.from_template(PRODUCT_QNA_TEMPLATE) | get_llm() | StrOutputParser()

    return chain.invoke({
        "title": product_title,
        "bullets": "\n".join(bullet_points),
        "description": description,
        "question": question,
